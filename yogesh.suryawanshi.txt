git init //Command in a folder where actual repository is located to initialize the git

git add . //add current folders files to git

get status //list of files needs to be add

git commit -m "comment" //commit files

git config --global user.email "email id"

git config --global user.name "your name"

git //list all the commands on command line



git remote add original git@github.com:atulp-iprogrammer/gitpractice.git  //add the other repo

git remove -v // display origin -- long url

git push origin master //master - default ------- This command to push changes to repo. --- may give error if changes are not updated

git pull origin master  //Pull changes to local from repo.

git checkout -b dev // Create branch dev

git add // Add changes again.

git commit -am "message"   // add and commit.

git checkout master // Switch branch to master

git merge dev //Merge dev branch into master

git diff // to show difference.

git checkout filename // to discard changes to the file.

git branch -d //Delete the branch if branch is merged is 

git rm filename //Delete the file and will be added into git ignore

.gitignore //Keep the files in this file which not require to be committed.

git reset <checksum number> // revert to revision of check sum number -- Hard to delete hard

git stash //move files to temp location

git apply //Move back files from temporary files
